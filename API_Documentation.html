üìã Overview

This API provides a comprehensive venue booking system for various activities like football, swimming, rock climbing, fun parks, etc. The system supports:

  • üéØ Multi-venue management with different activities
  • üí∞ Flexible pricing with categories (Adult/Child)
  • üìÖ Time slot booking with availability checking
  • üí≥ Payment processing with partial payment support
  • üë• Role-based access control (Admin, Owner, User)
  • üîê External authentication integration
Base URL: Your .NET API base URL (e.g., https://your-api.com)
External Auth URL: https://bandobasta-latest-5u7o.onrender.com/bandobasta/api/v1

1. Authentication Service

// services/authService.js
class AuthService {
  constructor() {
    this.baseUrl = 'https://bandobasta-latest-5u7o.onrender.com/bandobasta/api/v1';
    this.apiUrl = 'YOUR_DOTNET_API_BASE_URL';
  }

  async login(identifier, password) {
    try {
      const response = await fetch(`${this.baseUrl}/user/authenticate/login`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ identifier, password })
      });

      if (!response.ok) {
        throw new Error('Login failed');
      }

      const result = await response.json();
      
      if (result.code === '200') {
        // Store token and user info
        localStorage.setItem('authToken', result.data.accessToken);
        localStorage.setItem('userId', result.data.id);
        localStorage.setItem('userRoles', JSON.stringify(result.data.roles));
        
        // Get detailed user info
        await this.getUserDetails(result.data.id);
        
        return result.data;
      } else {
        throw new Error(result.message || 'Login failed');
      }
    } catch (error) {
      console.error('Login error:', error);
      throw error;
    }
  }

  async getUserDetails(userId) {
    try {
      const token = localStorage.getItem('authToken');
      const response = await fetch(`${this.baseUrl}/user/${userId}`, {
        headers: {
          'Authorization': `Bearer ${token}`
        }
      });

      if (!response.ok) {
        throw new Error('Failed to get user details');
      }

      const result = await response.json();
      
      if (result.code === '200') {
        localStorage.setItem('userDetails', JSON.stringify(result.data));
        return result.data;
      }
    } catch (error) {
      console.error('Get user details error:', error);
    }
  }

  logout() {
    localStorage.removeItem('authToken');
    localStorage.removeItem('userId');
    localStorage.removeItem('userRoles');
    localStorage.removeItem('userDetails');
  }

  isAuthenticated() {
    return !!localStorage.getItem('authToken');
  }

  getUserRoles() {
    const roles = localStorage.getItem('userRoles');
    return roles ? JSON.parse(roles) : [];
  }

  hasRole(role) {
    const roles = this.getUserRoles();
    return roles.includes(role);
  }

  getAuthHeaders() {
    const token = localStorage.getItem('authToken');
    return {
      'Authorization': `Bearer ${token}`,
      'Content-Type': 'application/json'
    };
  }
}

export default new AuthService();

2. API Service

// services/apiService.js
import authService from './authService';

class ApiService {
  constructor() {
    this.baseUrl = 'YOUR_DOTNET_API_BASE_URL';
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseUrl}${endpoint}`;
    
    const config = {
      ...options,
      headers: {
        ...options.headers,
        ...authService.getAuthHeaders()
      }
    };

    try {
      const response = await fetch(url, config);
      
      if (!response.ok) {
        if (response.status === 401) {
          // Token expired or invalid
          authService.logout();
          window.location.href = '/login';
          return;
        }
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      return await response.json();
    } catch (error) {
      console.error('API request error:', error);
      throw error;
    }
  }

  // Venues
  async getVenues() {
    return this.request('/api/venues');
  }

  async getVenue(id) {
    return this.request(`/api/venues/${id}`);
  }

  async createVenue(venueData) {
    return this.request('/api/venues', {
      method: 'POST',
      body: JSON.stringify(venueData)
    });
  }

  async updateVenue(id, venueData) {
    return this.request(`/api/venues/${id}`, {
      method: 'PUT',
      body: JSON.stringify(venueData)
    });
  }

  async deleteVenue(id) {
    return this.request(`/api/venues/${id}`, {
      method: 'DELETE'
    });
  }

  async getMyVenues() {
    return this.request('/api/venues/my-venues');
  }

  // Bookings
  async createBooking(bookingData) {
    return this.request('/api/bookings', {
      method: 'POST',
      body: JSON.stringify(bookingData)
    });
  }

  async getBooking(id) {
    return this.request(`/api/bookings/${id}`);
  }

  async getUserBookings(userId) {
    return this.request(`/api/bookings/user/${userId}`);
  }

  async cancelBooking(id, reason) {
    return this.request(`/api/bookings/${id}/cancel?reason=${encodeURIComponent(reason)}`, {
      method: 'POST'
    });
  }

  // Payments
  async processPayment(paymentData) {
    return this.request('/api/payments/process', {
      method: 'POST',
      body: JSON.stringify(paymentData)
    });
  }

  // Availability
  async checkAvailability(params) {
    const queryString = new URLSearchParams(params).toString();
    return this.request(`/api/availability?${queryString}`);
  }
}

export default new ApiService();

    // components/ProtectedRoute.jsx
import React from 'react';
import { useAuth } from '../hooks/useAuth';
import { Navigate } from 'react-router-dom';

const ProtectedRoute = ({ children, requiredRoles = [] }) => {
  const { isAuthenticated, hasRole } = useAuth();

  if (!isAuthenticated) {
    return ;
  }

  if (requiredRoles.length > 0) {
    const hasRequiredRole = requiredRoles.some(role => hasRole(role));
    if (!hasRequiredRole) {
      return ;
    }
  }

  return children;
};

export default ProtectedRoute;

    useAuth Hook

    // hooks/useAuth.js
import { useState, useEffect, createContext, useContext } from 'react';
import authService from '../services/authService';

const AuthContext = createContext();

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within an AuthProvider');
  }
  return context;
};

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(null);
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const initAuth = async () => {
      try {
        if (authService.isAuthenticated()) {
          const userDetails = localStorage.getItem('userDetails');
          if (userDetails) {
            setUser(JSON.parse(userDetails));
            setIsAuthenticated(true);
          }
        }
      } catch (error) {
        console.error('Auth initialization error:', error);
        authService.logout();
      } finally {
        setLoading(false);
      }
    };

    initAuth();
  }, []);

  const login = async (identifier, password) => {
    try {
      const userData = await authService.login(identifier, password);
      setUser(userData);
      setIsAuthenticated(true);
      return userData;
    } catch (error) {
      throw error;
    }
  };

  const logout = () => {
    authService.logout();
    setUser(null);
    setIsAuthenticated(false);
  };

  const hasRole = (role) => {
    return authService.hasRole(role);
  };

  const value = {
    user,
    isAuthenticated,
    loading,
    login,
    logout,
    hasRole
  };

  return (
    
      {children}
    
  );
};
